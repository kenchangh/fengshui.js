Fengshui.js

Making contextual jQuery suck less


var todos = new fengshui.System({
  root: 'todo',
  types: {
    'value': ['newTodo']
  }
});

todos.todoInput.keypress(function(e) {
  var todo = todos.setCtx(this); // setting todo context
  if (e.which === 13) {
    // exit early if empty
    if (/\S+/.test(todo.newTodo)) return;
    var newTodo = '<div class="">'+todo.newTodo+'</div>';
    todo.todoList.append(todo.newTodo);
  }
});
